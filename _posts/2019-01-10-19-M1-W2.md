#### 1、python操作Excel读写--使用xlrd
```python
使用实战

    def open_excel(self, file):  # 打开要解析的Excel文件
        try:
            data = xlrd.open_workbook(file)
            return data
        except Exception as e:
            self.log.error('open file %s failed. ' % file)
            self.log.error(str(e))
    
    def read_excel(self, file, by_index=0):
        """
        读取文件数据，需要上传者严格保证文件格式，程序未做格式校验
        :param file:
        :param by_index:
        :return:
        """
        columns = [
            'yx_spuId',
            'yx_skuId',
            'name',
            'sku_desc',
            'original_price',
            'daily_price',
            'promotional_price',
            'cost_price']
        data = self.open_excel(file)
        tab = data.sheets()[by_index]  # 选择excel里面的Sheet，默认第一个
        nrows = tab.nrows  # 行数
        ncols = tab.ncols  # 列数
        goods_list = []
        for x in range(1, nrows):  
            row = tab.row_values(x)
            if row:
                goods = dict()
                for y in range(0, min(ncols, 8)): # 只取前8列，忽略多余的列
                    goods[columns[y]] = row[y]
                goods_list.append(goods)
        return goods_list
```

```python
import os

basedir = os.path.abspath(os.path.dirname(__file__))

    pass

config = {
    'testing': TestingConfig,

    'default': TestingConfig
}

2、引用配置文件

from conf.config import config

def get_config():
    return config['default']

from core import get_config
```

```python
示例代码:
from fake_useragent import UserAgent
import requests

ua=UserAgent()
#请求的网址
url="http://www.baidu.com"
#请求头
headers={"User-Agent":ua.random}
#请求网址
response=requests.get(url=url,headers=headers)
#响应体内容
print(response.text)
#响应状态信息
print(response.status_code)
#响应头信息
print(response.headers)
```
